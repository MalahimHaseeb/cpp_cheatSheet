Logical Operators

                    
            #include<iostream>
            using namespace std;

            int main() {
            bool condition1 = true, condition2= false;
            bool logicalAndResult = condition1 && condition2;
            bool logicalOrResult = condition1 || condition2;
            bool logicalNotResult = !condition1;

            cout<<"The result of logicalAndResult is " <<logicalAndResult<<endl ;
            cout<<"The result of logicalOrResult is " <<logicalOrResult<<endl ;
            cout<<"The result of logicalNotResult is " <<logicalNotResult<<endl ;
            return 0;
            }

Passing Array to function

                    
            #include<iostream>
            using namespace std;

            void PrintArray(int arr[] , int size){
                for(int i = 0 ; i< size ; i++){
                    cout<< arr[i]<<" " ;
                }

            }

            int main() {
                int nums[5] = {1,2,3,4,5};

            PrintArray(nums , 5);
            return 0;
            }
